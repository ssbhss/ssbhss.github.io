2.编写配置文件

2.1 SpringConfig

@Configuration
@ComponentScan(basePackages = "org.example.service")
@PropertySource("classpath:jdbc.properties")
@Import({JdbcConfig.class, MybatisConfig.class})
public class SpringConfig {
}

2.2 SpringMvcConfig

@Configuration
@ComponentScan("org.example.controller")
@EnableWebMvc
public class SpringMvcConfig {
}

2.3 JdbcConfig

public class JdbcConfig {

    @Value("${jdbc.driver}")
    private String driverClassName;
    @Value("${jdbc.url}")
    private String url;
    @Value("${jdbc.username}")
    private String username;
    @Value("${jdbc.password}")
    private String password;
    @Bean
    public DataSource dataSource() {
        DruidDataSource dataSource = new DruidDataSource();
        dataSource.setDriverClassName(driverClassName);
        dataSource.setUrl(url);
        dataSource.setUsername(username);
        dataSource.setPassword(password);
        return dataSource;
    }
}

2.4MybatisConfig

public class MybatisConfig {

    @Bean
    public SqlSessionFactoryBean sqlSessionFactory(DataSource dataSource){
        SqlSessionFactoryBean factoryBean = new SqlSessionFactoryBean();
        factoryBean.setDataSource(dataSource);
        factoryBean.setTypeAliasesPackage("org.example.domain");
        return factoryBean;
    }

    @Bean
    public MapperScannerConfigurer mapperScannerConfigurer(){
        MapperScannerConfigurer mapperScannerConfigurer = new MapperScannerConfigurer();
        mapperScannerConfigurer.setBasePackage("org.example.dao");
        return mapperScannerConfigurer;
    }
}

2.5 ServletConfig

public class ServletConfig extends AbstractAnnotationConfigDispatcherServletInitializer {
    protected Class<?>[] getRootConfigClasses() {
        return new Class[]{SpringConfig.class};
    }

    protected Class<?>[] getServletConfigClasses() {
        return new Class[]{SpringMvcConfig.class};
    }

    protected String[] getServletMappings() {
        return new String[]{"/"};
    }
}

3. 编写domain层代码

@Data
public class Book {

    public int id;
    public String type;
    public String name;
    public String description;
}

4. 编写dao层代码

public interface BookDao {

    @Select("select * from tbl_book")
    List<Book> selectAll();

    @Select("select * from tbl_book where id=#{id}")
    Book selectById(int id);

    @Delete("delete from tbl_book where id =#{id}")
    void deleteById(int id);

    @Update("update tbl_book set id=#{id}, type=#{type}, name=#{name}, description=#{description} where id=#{id}")
    void update(Book book);
    @Insert("insert into tbl_book values (null, #{type}, #{name}, #{description})")
    void save(Book book);
}

5. 编写service层接口和实现类

5.1 接口

public interface BookService {
    
    List<Book> selectAll();
    
    Book selectById(int id);

    boolean deleteById(int id);

    boolean update(Book book);

    boolean save(Book book);
}

5.2 实现类

@Service
public class BoockServiceImpl implements BookService {

    @Autowired
    private BookDao bookDao;
    public List<Book> selectAll() {
        return bookDao.selectAll();
    }

    public Book selectById(int id) {
        return bookDao.selectById(id);
    }

    public boolean deleteById(int id) {
        bookDao.deleteById(id);
        return true;
    }

    public boolean update(Book book) {
        bookDao.update(book);
        return true;
    }

    public boolean save(Book book) {
        bookDao.save(book);
        return true;
    }
}

6. 编写controller层代码

@RestController
@RequestMapping("/books")
@ResponseBody
public class BookController {

    @Autowired
    BookService bookService;

    @PostMapping
    public boolean save(@RequestBody Book book){
        System.out.println("save...");
        return bookService.save(book);
    }

    @PutMapping
    public boolean update(@RequestBody Book book){
        System.out.println("update...");
        return bookService.update(book);
    }

    @DeleteMapping(path="/{id}")
    public boolean deleteById(@PathVariable int id){
        System.out.println("delete...");
        return bookService.deleteById(id);
    }

    @GetMapping(path="/{id}")
    public Book selectById(@PathVariable int id){
        System.out.println("select by Id...");
        return bookService.selectById(id);
    }

    @GetMapping
    public List<Book> selectAll(){
        System.out.println("selectAll...");
        return bookService.selectAll();
    }
}

8. 总结

一套基于Spring的SSM开发的基本流程如下。首先通过Maven导入需要使用的jar包，然后配置Jdbc的连接信息。接下来在config包下配置Spring、SpringMvc、Mybatis、Servlet、Jdbc的配置信息。完成后，根据数据库中的表创建POJO类，并完成dao曾的接口。紧接着，根据业务要求创建service接口，并实例化出实现业务的实现类。值得注意的是，service实现类中自动装配的dao曾bean对象是来自于Mapper代理类。最后，通过controller使用Restful风格完成地址映射。
